import hashlib
from pathlib import Path
import chromadb
from langchain.text_splitter import RecursiveCharacterTextSplitter

# 1. Инициализация базы (локально, экономно для RAM)
client = chromadb.PersistentClient(path="./vector_storage")
collection = client.get_or_create_collection(name="user_knowledge_base")

def get_file_hash(file_path):
    """Считает хеш файла, не загружая его целиком в память."""
    hash_sha256 = hashlib.sha256()
    with open(file_path, "rb") as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hash_sha256.update(chunk)
    return hash_sha256.hexdigest()

def check_in_vector_db(file_id):
    """Проверяет, есть ли хотя бы один чанк с таким file_id."""
    results = collection.get(
        where={"file_id": file_id},
        limit=1,
        include=[] # Важно: не грузим данные, только проверяем наличие
    )
    return len(results['ids']) > 0

async def process_file_to_db(file_path, user_id):
    """Основная функция: Хеш -> Проверка -> Загрузка -> Сплиттер -> Сохранение."""
    
    # Шаг 1: Быстрая проверка на дубликат по хешу
    file_id = get_file_hash(file_path)
    if check_in_vector_db(file_id):
        return f"Файл уже есть в вашей базе (ID: {file_id[:8]}...)"

    # Шаг 2: Подбор лоадера и извлечение текста
    loader = await get_loader_for_file(file_path) # Ваша функция из прошлых сообщений
    if isinstance(loader, str): return loader # Ошибка формата
    
    try:
        docs = loader.load()
        raw_text = "\n".join([doc.page_content for doc in docs])
    except Exception as e:
        return f"Ошибка при чтении файла: {e}"

    # Шаг 3: Нарезка текста на чанки
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=800,
        chunk_overlap=100
    )
    chunks = text_splitter.split_text(raw_text)

    # Шаг 4: Подготовка метаданных и ID
    ids = [f"{file_id}_{i}" for i in range(len(chunks))]
    metadatas = [
        {
            "file_id": file_id, 
            "user_id": user_id, 
            "filename": Path(file_path).name
        } for _ in range(len(chunks))
    ]

    # Шаг 5: Сохранение в базу
    collection.add(
        ids=ids,
        documents=chunks,
        metadatas=metadatas
    )

    return f"Успешно! Файл разбит на {len(chunks)} частей и добавлен в базу."

# Функция удаления для пользователя
def delete_user_file(file_id, user_id):
    """Удаляет все части файла, проверяя, что он принадлежит пользователю."""
    collection.delete(
        where={
            "$and": [
                {"file_id": file_id},
                {"user_id": user_id}
            ]
        }
    )
    return "Файл удален."
